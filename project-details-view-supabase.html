To use this app with Supabase, follow these steps:

  1. Create a free account at supabase.com
  2. Create a new project
  3. In the SQL Editor, run this SQL: 
    CREATE TABLE projects (
  id BIGSERIAL PRIMARY KEY,
  project_name TEXT NOT NULL,
  foundation_team TEXT,
  project_types JSONB,
  raisers_edge_info JSONB,
  items JSONB,
  event_details JSONB,
  discount_codes JSONB,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);

ALTER TABLE projects ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Enable all access" ON projects FOR ALL USING (true);

  5. Copy your URL and anon key
